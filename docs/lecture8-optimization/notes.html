import pandas

def call_js(f):
  if not callable(f):
    return f

  def call(*args):
    args = [(a.to_dict('records') if isinstance(a, pandas.DataFrame) else a) for a in args]
    args = [to_js(a, dict_converter=Object.fromEntries) for a in args]

def sigmoid(x):
    return 1 / (1 + np.exp(-x))

def tangent(f, x):
    y = f(x)
    slope = grad(f)(x)

    def tf(a):
        return (a - x) * slope + y

    return tf

def tangent2(f, x):
    y = f(x)
    slope = grad(f)(x)
    curve = 0.5 * grad(grad(f))(x)

    def tf(a):
        return curve * (a - x) ** 2 + (a - x) * slope + y
    return tf
